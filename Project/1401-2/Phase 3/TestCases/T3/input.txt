void main(void)
{
	int a[2];
	int b[2];
    int i;
    int k;
	b[0] = 4;
	b[1] = 5;

	a[0] = 2;
	a[1] = 3;

	a[0] = a[0]*b[0];
	a[1] = a[1]+b[1];

	i = 0;
	repeat{
    output(a[i]);
    i = i + 1;
    }
	until(1 < i)

	k = 10;
	repeat{
    if(k == 6){
        break;
    }
    else{
        output(k+1);
        k = k - 1;
    }
    }
    until(k < 0)
}
